
// ÖRNEK Firestore Kuralları
// Firebase Console -> Firestore -> Rules kısmına uyarlaman gerekir.
// Burada basit bir örnek veriyorum, projene göre düzenlemelisin.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Kullanıcı başvuruları
    match /users/{userId} {
      allow read, write: if request.auth != null; // Geçici: Tüm authenticated kullanıcılar
    }

    // Roller koleksiyonu - sadece admin panelinden yönetilmeli
    match /roles/{uid} {
      allow read: if request.auth != null && request.auth.uid == uid;
      allow write: if false; // Elle konsoldan yönet
    }
  }
}
